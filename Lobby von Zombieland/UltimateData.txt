-- Helper Functions
function UltimateData:GetUltimate(name)
	-- Check regular ultimates
	for epoch, ultimates in pairs(self.Ultimates) do
		for _, ultimate in ipairs(ultimates) do
			if ultimate.name == name then
				return ultimate, epoch
			end
		end
	end

	-- Check special ultimates
	for _, ultimate in ipairs(self.SpecialUltimates) do
		if ultimate.name == name then
			return ultimate, "Special"
		end
	end

	return nil
end

function UltimateData:GetRarityConfig(rarity)
	return self.RarityConfig[rarity] or self.RarityConfig.Common
end

return UltimateData
